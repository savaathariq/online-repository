    public void Hitung(float input){
        
        fahren = (float) ((1.8*input)+32);
        reamur = (float) (0.8 * input);
        kelvin = (float) (input + 273.15);
        
       
        txtFahrenheit.setText("" + fahren);
        txtReamur.setText("" + reamur);
        txtKelvin.setText("" + kelvin);
    }
